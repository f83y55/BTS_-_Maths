  2. Les acteurs et les films disposant chacun d'une table propre, la table des rôles fait référence à deux attributs de ces tables.

    ALTER TABLE Joue_Role
ADD CONSTRAINT Idacteur,
FOREIGN KEY Idacteur REFERENCES Acteur.Idacteur

  4. SELECT realisateur_nom, realisateur_prenom FROM Realisateur
